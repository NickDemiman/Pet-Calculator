177 defineIterationElements(ref prev, ref elem, ref next, expression, i);
178

182 {

184 number.Insert(0, elem);

187 FormNumber(ref number, ref result);
188 PushOperatorToStack(ref result, ref op_queue, elem);
189 }
190 }
191 else
192 {
193 FormNumber(ref number, ref result);
194 PushOperatorToStack(ref result, ref op_queue, elem);
195 }
196 }
197 else
198 {
199 number.Append(elem);
200 }
201 }
202

205 }
